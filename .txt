import 'dart:async';
import 'package:flutter/material.dart';

void main() {
  runApp(const MainApp());
}

class MainApp extends StatefulWidget {
  const MainApp({super.key});

  @override
  State<MainApp> createState() => _MainAppState();
}

class _MainAppState extends State<MainApp> {
  int skor = 0;
  int kalansure = 10;
  bool oyundurum = false;
  Timer? zamanlayici;

  void baslat() {
    setState(() {
      skor = 0;
      kalansure = 10;
      oyundurum = true;
    });

    zamanlayici?.cancel();

    zamanlayici = Timer.periodic(const Duration(seconds: 1), (timer) {
      setState(() {
        if (kalansure > 0) {
          kalansure--;
        } else {
          oyundurum = false;
          zamanlayici?.cancel();
        }
      });
    });
  }

  void tikla() {
    if (oyundurum) {
      setState(() {
        skor++;
      });
    }
  }

  @override
  void dispose() {
    zamanlayici?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(
          centerTitle: true,
          title: const Text("CLICKER GAME"),
          backgroundColor: Colors.amber,
        ),
        body: Center(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              Text(
                "Skor: $skor",
                style: const TextStyle(
                  fontSize: 24,
                  fontWeight: FontWeight.bold,
                ),
              ),
              const SizedBox(height: 20),
              Text(
                oyundurum ? "Kalan süre: $kalansure" : "Oyun Bitti!",
                style: const TextStyle(
                  fontSize: 30,
                  color: Colors.red,
                  fontWeight: FontWeight.bold,
                ),
              ),
              const SizedBox(height: 40),
              IconButton(
                onPressed: tikla,
                icon: const Icon(Icons.bolt),
                iconSize: 80,
                color: Colors.white,
                style: IconButton.styleFrom(
                  backgroundColor: oyundurum ? Colors.blue : Colors.grey,
                  padding: const EdgeInsets.all(40),
                ),
              ),
              const SizedBox(height: 40),
              ElevatedButton(
                onPressed: oyundurum ? null : baslat,
                child: const Text("Oyunu Başlat"),
              ),
            ],
          ),
        ),
      ),
    );
  }
}
